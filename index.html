<script>

    let words = JSON.parse(localStorage.getItem('myVocab')) || [];


    const cnInput = document.getElementById('cn-input');

    const enInput = document.getElementById('en-input');

    const addBtn = document.getElementById('add-btn');

    const vocabList = document.getElementById('vocab-list');


    // 初始化渲染

    renderList();


    // 添加单词逻辑

    addBtn.addEventListener('click', () => {

        const cn = cnInput.value.trim();

        const en = enInput.value.trim();


        if (cn && en) {

            words.push({ cn, en, count: 0, time: Date.now() });

            saveAndRender();

            cnInput.value = '';

            enInput.value = '';

            enInput.focus();

        }

    });


    // 增加计数并排序

    function incrementCount(index) {

        words[index].count += 1;

        saveAndRender();

    }


    function saveAndRender() {

        // 核心排序逻辑：按点击次数（学习频率）从高到低排序

        words.sort((a, b) => b.count - a.count);

        

        localStorage.setItem('myVocab', JSON.stringify(words));

        renderList();

    }


    function renderList() {

        vocabList.innerHTML = '';

        words.forEach((item, index) => {

            const li = document.createElement('li');

            li.className = 'vocab-item';

            li.onclick = () => incrementCount(index);


            li.innerHTML = `

                <div class="vocab-content">

                    <span class="cn">${item.cn}</span>

                    <span class="en">${item.en}</span>

                </div>

                <span class="count-badge">已学 ${item.count} 次</span>

            `;

            vocabList.appendChild(li);

        });

    }


    // 回车添加功能

    enInput.addEventListener('keypress', (e) => {

        if (e.key === 'Enter') addBtn.click();

    });

</script>
